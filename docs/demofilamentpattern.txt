  <script>
    // Элементы
    const preview = document.getElementById('preview');
    const rodBody = document.getElementById('rodBody');
    const rodEnd = document.getElementById('rodEnd');
    const fillerOverlay = document.getElementById('fillerOverlay');
    const fillerOverlayEnd = document.getElementById('fillerOverlayEnd');
    
    const color1Input = document.getElementById('color1');
    const color2Input = document.getElementById('color2');
    const color3Input = document.getElementById('color3');
    const color4Input = document.getElementById('color4');
    const color5Input = document.getElementById('color5');
    
    const colorTypeRadios = document.querySelectorAll('input[name="colorType"]');
    const finishRadios = document.querySelectorAll('input[name="finish"]');
    const fillerRadios = document.querySelectorAll('input[name="filler"]');
    const transparencySlider = document.getElementById('transparencySlider');
    const transparencyFill = document.getElementById('transparencyFill');
    const transparencyHandle = document.getElementById('transparencyHandle');
    const transparencyValue = document.getElementById('transparencyValue');
    
    let transparency = 50; // по умолчанию 50%
    
    // Обновление превью
    function updatePreview() {
      // Сброс классов
      rodBody.className = 'rod-body';
      rodEnd.className = 'rod-end';
      fillerOverlay.className = 'filler-overlay';
      fillerOverlayEnd.className = 'filler-overlay';
      
      // Получение значений
      const color1 = color1Input.value;
      const color2 = color2Input.value;
      const color3 = color3Input.value;
      const color4 = color4Input.value;
      const color5 = color5Input.value;
      
      const colorType = document.querySelector('input[name="colorType"]:checked').id;
      const finish = document.querySelector('input[name="finish"]:checked').id;
      const filler = document.querySelector('input[name="filler"]:checked').id;
      
      // Применение финиша
      if (finish === 'glossy') {
        preview.classList.add('glossy');
      } else {
        preview.classList.remove('glossy');
      }
      
      // Цвета
      if (colorType === 'single') {
        rodBody.style.background = color1;
        rodEnd.style.background = color1;
      } 
      else if (colorType === 'two') {
        rodBody.style.background = `linear-gradient(to bottom, ${color1} 50%, ${color2} 50%)`;
        rodEnd.style.background = `conic-gradient(from 0deg, ${color1} 0% 50%, ${color2} 50% 100%)`;
      }
      else if (colorType === 'three') {
        rodBody.style.background = `linear-gradient(to bottom, ${color1} 33.3%, ${color2} 33.3% 66.6%, ${color3} 66.6%)`;
        rodEnd.style.background = `conic-gradient(from 0deg, ${color1} 0% 33.3%, ${color2} 33.3% 66.6%, ${color3} 66.6% 100%)`;
      }
      else if (colorType === 'gradient') {
        // Многоцветный градиент — используем все цвета
        const colors = [color1, color2, color3, color4, color5];
        const steps = [];
        const step = 100 / (colors.length - 1);
        
        for (let i = 0; i < colors.length; i++) {
          steps.push(`${colors[i]} ${i * step}%`);
        }
        
        const gradient = `linear-gradient(to right, ${steps.join(', ')})`;
        rodBody.style.background = gradient;
        rodEnd.style.background = color5; // последний цвет на конце
      }
      
      // Применение наполнителя
      if (filler !== 'none') {
        fillerOverlay.classList.add(`${filler}-texture`);
        fillerOverlayEnd.classList.add(`${filler}-texture`);
        
        // Применяем прозрачность к наполнителю
        const opacity = transparency / 100;
        fillerOverlay.style.opacity = opacity;
        fillerOverlayEnd.style.opacity = opacity;
      }
      
      // Применение прозрачности к основному фону
      if (transparency < 100) {
        rodBody.style.background = `linear-gradient(135deg, rgba(${parseInt(color1.slice(1,3), 16)}, ${parseInt(color1.slice(3,5), 16)}, ${parseInt(color1.slice(5,7), 16)}, ${transparency/100}), rgba(${parseInt(color1.slice(1,3), 16)}, ${parseInt(color1.slice(3,5), 16)}, ${parseInt(color1.slice(5,7), 16)}, ${transparency/100}))`;
        rodEnd.style.background = `linear-gradient(135deg, rgba(${parseInt(color1.slice(1,3), 16)}, ${parseInt(color1.slice(3,5), 16)}, ${parseInt(color1.slice(5,7), 16)}, ${transparency/100}), rgba(${parseInt(color1.slice(1,3), 16)}, ${parseInt(color1.slice(3,5), 16)}, ${parseInt(color1.slice(5,7), 16)}, ${transparency/100}))`;
      }
    }
    
    // События
    color1Input.addEventListener('input', updatePreview);
    color2Input.addEventListener('input', updatePreview);
    color3Input.addEventListener('input', updatePreview);
    color4Input.addEventListener('input', updatePreview);
    color5Input.addEventListener('input', updatePreview);
    
    colorTypeRadios.forEach(radio => radio.addEventListener('change', updatePreview));
    finishRadios.forEach(radio => radio.addEventListener('change', updatePreview));
    fillerRadios.forEach(radio => radio.addEventListener('change', updatePreview));
    
    // Ползунок прозрачности
    transparencySlider.addEventListener('mousedown', (e) => {
      const startX = e.clientX;
      const startLeft = parseFloat(window.getComputedStyle(transparencyHandle).left);
      
      const moveHandler = (moveEvent) => {
        const deltaX = moveEvent.clientX - startX;
        let newLeft = startLeft + deltaX;
        if (newLeft < 0) newLeft = 0;
        if (newLeft > 184) newLeft = 184; // ширина слайдера - размер ручки
        
        transparency = Math.round((newLeft / 184) * 100);
        transparencyFill.style.width = `${transparency}%`;
        transparencyHandle.style.left = `${newLeft}px`;
        transparencyValue.textContent = `${transparency}%`;
        updatePreview();
      };
      
      const upHandler = () => {
        document.removeEventListener('mousemove', moveHandler);
        document.removeEventListener('mouseup', upHandler);
      };
      
      document.addEventListener('mousemove', moveHandler);
      document.addEventListener('mouseup', upHandler);
    });
    
    // Инициализация
    updatePreview();
  </script>
</body>
